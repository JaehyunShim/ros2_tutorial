32 namespace action_tutorial_cpp
33 {
34 class ActionServer : public rclcpp::Node // public inherit from rclcpp::Node
35 {
36 public:
37  ActionServer()
38  : Node("action_server"), // initialize the Node with the name "action_server"
39  goal_received_(false),
40  action_accepted_(false),
41  action_canceled_(false)
42  {
43  this->action_server_ = rclcpp_action::create_server<Fibonacci>(
44  this,
45  "fibonacci",
46  std::bind(&ActionServer::handle_goal, this, _1, _2),
47  std::bind(&ActionServer::handle_cancel, this, _1),
48  std::bind(&ActionServer::handle_accepted, this, _1));
49  }
50 
51  bool goal_received_;
52  bool action_accepted_;
53  bool action_canceled_;
54 
55 private:
56  rclcpp_action::Server<Fibonacci>::SharedPtr action_server_;
57  rclcpp_action::GoalResponse handle_goal(
58  const rclcpp_action::GoalUUID & uuid, // https://github.com/mariusbancila/stduuid
59  std::shared_ptr<const Fibonacci::Goal> goal)
60  {
61  RCLCPP_INFO(this->get_logger(), "Received goal request with order %d", goal->order);
62  (void)uuid; // not used
63  goal_received_ = true;
64  return rclcpp_action::GoalResponse::ACCEPT_AND_EXECUTE;
65  }
66  rclcpp_action::CancelResponse handle_cancel(
67  const std::shared_ptr<ServerGoalHandleFibonacci> goal_handle)
68  {
69  RCLCPP_INFO(this->get_logger(), "Received request to cancel goal");
70  (void)goal_handle;
71  action_canceled_ = true;
72  return rclcpp_action::CancelResponse::ACCEPT;
73  }
74  void handle_accepted(const std::shared_ptr<ServerGoalHandleFibonacci> goal_handle)
75  {
76  std::thread(std::bind(&ActionServer::execute, this, _1), goal_handle).detach();
77  action_accepted_ = true;
78  }
79 
80  void execute(const std::shared_ptr<ServerGoalHandleFibonacci> goal_handle)
81  {
82  RCLCPP_INFO(this->get_logger(), "Executing goal");
83  rclcpp::Rate loop_rate(1);
84  const auto goal = goal_handle->get_goal();
85  auto feedback = std::make_shared<Fibonacci::Feedback>();
86  auto & sequence = feedback->partial_sequence;
87  sequence.push_back(0);
88  sequence.push_back(1);
89  auto result = std::make_shared<Fibonacci::Result>();
90 
91  for (int i = 1; (i < goal->order) && rclcpp::ok(); ++i) {
92  // Check if there is a cancel request
93  if (goal_handle->is_canceling()) {
94  result->sequence = sequence;
95  goal_handle->canceled(result);
96  RCLCPP_INFO(this->get_logger(), "Goal canceled");
97  return;
98  }
99 
100  // Update sequence
101  sequence.push_back(sequence[i] + sequence[i - 1]);
102  // Publish feedback
103  goal_handle->publish_feedback(feedback);
104  RCLCPP_INFO(this->get_logger(), "Publish feedback");
105 
106  loop_rate.sleep();
107  }
108 
109  // Check if goal is done
110  if (rclcpp::ok()) {
111  result->sequence = sequence;
112  goal_handle->succeed(result);
113  RCLCPP_INFO(this->get_logger(), "Goal succeeded");
114  }
115  }
116 };
