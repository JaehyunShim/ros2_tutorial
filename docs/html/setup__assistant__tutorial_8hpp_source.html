53 std::string replace_line(std::string line, std::pair<std::string, std::string> pair)
54 {
55  std::string::size_type offset = 0;
56  offset = line.find(pair.first, offset);
57  if (offset != std::string::npos) {
58  line.replace(offset, pair.first.length(), pair.second);
59  }
60  return line;
61 }
